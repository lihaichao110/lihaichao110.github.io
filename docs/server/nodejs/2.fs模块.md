#### 文件写入
```js
// 文件写入
// 1. 导入工具模块
const fs = require('fs')

// 2. 写入文件（异步写入）
/**
 * 参数一： 要写入的文件路径和文件名
 * 参数二: 要写入的内容
 * 参数三: 回调函数
 * 返回值: undefined
 */
fs.writeFile('./demo.txt', 'hello world', (err) => {
  // err: 写入失败：错误对象
  // err: 写入成功：null
  if(err) return console.log(err.message)
  console.log('写入成功')
})

// 3. 写入文件（同步写入）
fs.writeFileSync('./demo1.txt', 'hello world123')
```

#### 追加写入
```js
// 追加写入
const fs = require('fs')

// 异步追加内容
fs.appendFile('./demo.txt', '\r\n 12345', (err) => {
  if(err) return console.log(err.message)
  console.log('追加成功')
})

// (可实现)异步追加内容
fs.writeFile('./demo.txt', '\r\n 12345', {flag: 'a'}, (err) => {
  if(err) return console.log(err.message)
  console.log('追加成功')
})

// 同步追加内容
fs.appendFileSync('./demo1.txt', '12345', (err) => {
  if(err) return console.log(err.message)
  console.log('追加成功')
})
```

#### 流式写入
```js
// 流式写入
// 导入 fs
const fs = require('fs')

// 创建写入流对象，填写写入路径
const ws = fs.createWriteStream('./观书有感.txt')

// 开始写入
ws.write('12435t\r\n');
ws.write('23456\r\n');
ws.write('456789\r\n');
ws.write('56789\r\n');

// 关闭写入流(可选)，会自动关闭
ws.close()
```

#### 文件读取
```js
// 流式写入
// 引入 fs 模块
const fs = require('fs')

// 异步读取文件
/**
 * 参数一: 要读取的文件路径
 * 参数二: 配置项（可选非必填）
 * 参数三: 回调函数
 */
fs.readFile('./观书有感.txt', 'utf-8', (err, data) => {
  if(err) return console.log(err.message)
  console.log(data)
})

// 同步读取文件
const data = fs.readFileSync('./观书有感.txt', 'utf-8')
console.log(data)
```
#### 流式读取
```js
// 流式读取
// 引入 fs 模块
const fs = require('fs')

// 创建流式读取对象
const rs = fs.createReadStream('./观书有感.txt')

// 绑定 data 事件，chunk 为每次读取的数据
rs.on('data', (chunk) => {
  console.log(chunk.length) // 65536 字节 => 64KB
  console.log(chunk.toString())
})

// 读取结束时触发
rs.on('end', () => {
  console.log('读取结束')
})
```

#### 练习 - 文件复制
```js
const fs = require('fs')

// 方式一：读取写入
let data = fs.readFileSync('./观书有感.txt', 'utf-8')
fs.writeFile('./观书有感2.txt', data, (err) => {
  if(err) return console.log(err.message)
  console.log('复制成功')
})

// 方式二：流式写入
let rs = fs.createReadStream('./观书有感.txt')
let ws = fs.createWriteStream('./观书有感3.txt')

rs.on('data', (chunk) => {
  ws.write(chunk)
})
```

#### 文件重命名与移动

const fs = require('fs')

#### 文件删除操作
```js
// 导入 fs 模块
const fs = require('fs')

// 调用 unlink 方法
// fs.unlinkSync(同步删除)
fs.unlink('./demo.txt', (err) => {
  if(err) return console.log(err.message)
  console.log('删除成功')
})

// 调用 rm 方法 14.4发布的
// fs.rmSync(同步删除)
fs.rm('./demo1.txt', (err) => {
  if(err) return console.log(err.message)
  console.log('删除成功')
})
```

#### 文件夹操作
```js
// 导入 fs 模块
const fs = require('fs')

// 创建文件夹 mk make 制作 dir directory 文件夹
/**
 * ? 参数一：要创建的文件夹路径
 * ? 参数二：配置项参数（可选非必填）
 * ? 参数三：回调函数
 */
// fs.mkdirSync() 同步创建
fs.mkdir('./html', (err) => {
  if(err) return console.log(err.message)
  console.log('创建成功')
})

// 递归创建文件夹，第二个参数要配置项
fs.mkdir('./a/b/c', {recursive: true} ,(err) => {
  if(err) return console.log(err.message)
  console.log('创建成功')
})


// 文件夹读取 read 读取 dir directory 文件夹
fs.readdir('../api_server', (err, files) => {
  if(err) return console.log(err.message)
  console.log(files)
})


// 删除文件夹 del 删除 dir directory 文件夹
fs.rmdir('./html', (err) => {
  if(err) return console.log(err.message)
  console.log('删除成功')
})

// 递归删除，消除警告，建议使用 rm 删除
fs.rm('./a', {recursive: true}, (err) => {
  if(err) return console.log(err.message)
  console.log('删除成功')
})
```
